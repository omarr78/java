``` java


  import java.util.*;
  
  public class Main {
  
      public static void main(String[] args) {
  
          String [] arr = {"abc", "123","456","xyz"};
  
          List<Integer> nums =  Arrays.stream(arr)
                  .filter(x -> x.matches("\\d+"))
                  .map(Integer::parseInt).toList();
          System.out.println(nums);
      }
  }
```
